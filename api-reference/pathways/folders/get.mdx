---
title: 'Get All Folders'
api: 'GET /v1/pathway/folders'
description: Retrieves all folders for the authenticated user, including folder ID, name, and parent folder ID.
---

### Headers

<ParamField header="authorization" type="string" required={true}>
  Your API key for authentication.
</ParamField>

### Response

<ResponseField name="folders" type="array">
   The unique identifier of the folder.
</ResponseField>
<ResponseField name="folders[].id" type="string">
  A message describing the result of the operation.
</ResponseField>
<ResponseField name="folders[].name" type="string">
  The name of the folder.
</ResponseField>
<ResponseField name="folders[].parent_folder_id" type="string">
  The ID of the parent folder, if applicable.
</ResponseField>

}

</ResponseExample>
