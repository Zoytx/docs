---
title: 'Get Inbound Campaign logs'
api: 'GET /api/campaigns/outbound/logs/{id}'
description: 'Retrieve a list of all existing campaigns'
---

### Headers

<ParamField header="Authorization" type="string" required>
  Your API key for authentication. Prefix with `Bearer `.
</ParamField>

<ParamField header="x-access-token" type="string" required>
  Your access token for authentication.
</ParamField>

### Response

<ResponseField name="data" type="object">
   The main response object containing analytics and logs.
</ResponseField>

<ResponseField name="data.analytics" type="object">
   An object containing overall analytics for the campaign logs.
</ResponseField>

<ResponseField name="data.analytics.Total Contacts" type="number">
   Total number of contacts involved in the campaign.
</ResponseField>

<ResponseField name="data.analytics.Total Conversations" type="number">
   Total number of conversations in the campaign.
</ResponseField>

<ResponseField name="data.logs" type="object[]">
   Array of log objects for each contact in the campaign.
</ResponseField>

<ResponseField name="data.logs.from" type="string">
   The sender's identifier or phone number.
</ResponseField>

<ResponseField name="data.logs.number" type="string">
   The recipient's phone number.
</ResponseField>

<ResponseField name="data.logs.status" type="string">
   Status of the campaign log (e.g., completed).
</ResponseField>

<ResponseField name="data.logs.call_sid" type="string">
   Identifier for the call session.
</ResponseField>

<ResponseField name="data.logs.sentiment" type="string">
   Sentiment analysis result for the call.
</ResponseField>

<ResponseField name="data.logs.inbound_id" type="string">
   Identifier for the inbound campaign.
</ResponseField>

<ResponseField name="data.logs.session_id" type="string">
   Identifier for the session.
</ResponseField>

<ResponseField name="data.logs.call_summary" type="string">
   Summary of the call conversation.
</ResponseField>

<ResponseField name="data.logs.call_duration" type="number">
   Duration of the call in seconds.
</ResponseField>

<ResponseField name="data.logs.goal_achieved" type="string">
   Indicates if the campaign goal was achieved ("yes" or "no").
</ResponseField>

<ResponseField name="data.logs.transcript" type="object">
   Object containing the transcript and recording URL.
</ResponseField>

<ResponseField name="data.logs.transcript.conversation" type="object[]">
   Array of conversation objects with role and content.
</ResponseField>

<ResponseField name="data.logs.transcript.conversation.role" type="string">
   Role in the conversation (user or assistant).
</ResponseField>

<ResponseField name="data.logs.transcript.conversation.content" type="string">
   Content of the conversation message.
</ResponseField>

<ResponseField name="data.logs.transcript.recording_url" type="string">
   URL to the transcript audio file.
</ResponseField>
